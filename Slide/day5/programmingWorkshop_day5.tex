    \subsection{物理タイマーの利用}
        \begin{frame}
            \frametitle{物理タイマーの利用}
            \begin{block}{物理タイマーの意義}
                \begin{itemize}
                    \item 決まった時間に決まった装置を通電できる
                    \item 最もシンプル
                    \item 間接照明，Arduinoなど用途様々（ラズパイは不可）
                \end{itemize}
            \end{block}
        \end{frame}
